\begin{align*} \dfrac{2}{3}-\dfrac{1}{6} \amp\amp\amp \text{LCD is \(6\)}\\ \dfrac{2\cdot 2}{2\cdot 3}-\dfrac{1}{6} \amp\amp\amp \text{Multiply first fraction by \(1=\dfrac{2}{2}\),}\\ \amp\amp\amp\text{the second already has a denominator of \(6\)}\\ \dfrac{4}{6}-\dfrac{1}{6} \amp\amp\amp \text{Same denominator, subtract numerators, \(4-1\)}\\ \dfrac{3}{6} \amp\amp\amp \text{Reduce answer, dividing numerator and denominator by \(3\)}\\ \dfrac{1}{2} \amp\amp\amp \text{Our Solution}\checkmark \end{align*}
